var green_loader = new Loader();
green_loader.showLoader(document.getElementById('GreenLoader'));
 
var red_loader = new Loader();
red_loader.setLoaderColor('Red');
red_loader.setAnimationDuration(5);
red_loader.setTextChange(["Loader finished"], 'last_show_after_end');
red_loader.showLoader(document.getElementById('RedLoader'));
 
var counterArray = new Array();
for (var i = 0; i < 100; i+=10){
counterArray.push(i.toString()+'%');}
counterArray.push('Now give me all your money');
var orange_loader = new Loader();
orange_loader.setLoaderColor('Orange');
orange_loader.setLoaderBackgroundColor('#eee');
orange_loader.setTextChange(counterArray, 'last_show_after_end');
orange_loader.showLoader(document.getElementById('OrangeLoader'));
 
var rgb_loader = new Loader();
rgb_loader.setLoaderColor('rgb(233,30,99)');
rgb_loader.setTextClass('className');
rgb_loader.setTextChange(["first string", "second string", 
                            "should appear before loader finished."]);
rgb_loader.showLoader(document.getElementById('rgbLoader'));
